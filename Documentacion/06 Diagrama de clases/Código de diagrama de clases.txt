' ==== CLASES ====

class Persona {
  - nombre : String
  - apellido : String
  - email : String
  - telefono : int

  + getNombre() : String
  + getApellido() : String
  + getEmail() : String
  + getTelefono() : int

  + setNombre(nombre : String)
  + setApellido(apellido : String)
  + setEmail(email : String)
  + setTelefono(telefono : int)
}

class Cliente {
  - idCliente : int
  
  + agregarCliente()
  + modificarCliente()
  + eliminarCliente()
}

class Profesional {
  - idProfesional : int
  - especialidad : String
  - agendaProfesional : List<Turno>

  + verAgenda() : List<Turno>
  + agregarProfesional()
  + modificarProfesional()
  + eliminarProfesional()
}

class Administrador {
  - telefonoAdministrador : int
}

class Turno {
  - fechaHora : DateTime
  - estado : String
  - servicio : Servicio

  + cambiarEstado(nuevoEstado : String)
  + consultarServicio() : Servicio
  + verificarAnticipacion()
  + agregarTurno()
  + modificarTurno()
  + eliminarTurno()
}

class Servicio {
  - tipo : String
  - precio : Float
  - duracion : String

  + getTipo() : String
  + getPrecio() : Float
  + setTipo(tipo : String)
  + setPrecio(precio : Float)
  + agregarServicio()
  + modificarServicio()
  + eliminarServicio()
}

' ==== CATALOGOS ====

class CatalogoClientes {
  - clientes : List<Cliente>
  + agregarCliente(c : Cliente)
  + eliminarCliente(c : Cliente)
  + buscarCliente(id : int) : Cliente
}

class CatalogoProfesionales {
  - profesionales : List<Profesional>
  + agregarProfesional(p : Profesional)
  + eliminarProfesional(p : Profesional)
  + buscarProfesional(id : int) : Profesional
}

class CatalogoServicios {
  - servicios : List<Servicio>
  + agregarServicio(s : Servicio)
  + eliminarServicio(s : Servicio)
  + buscarServicio(tipo : String) : Servicio
}

class CatalogoTurnos {
  - turnos : List<Turno>
  + agregarTurno(t : Turno)
  + eliminarTurno(t : Turno)
  + buscarTurno(fechaHora : DateTime) : Turno
}

class Controlador {
  - catalogoClientes : CatalogoClientes
  - catalogoProfesionales : CatalogoProfesionales
  - catalogoServicios : CatalogoServicios
  - catalogoTurnos : CatalogoTurnos

  + agregarProfesional(nombre : string, apellido : string)
  + eliminarProfesional(nombre : string, apellido : string)
  + modificarProfesional(nombre : string, apellido : string)

  + agregarTurno(fechaHora : dateTime, estado : string, servicio : string)
  + modificarTurno(fechaHora : dateTime, estado : string, servicio : string)
  + eliminarTurno(fechaHora : dateTime, estado : string, servicio : string)

  + agregarCliente(nombre : string, apellido : string, email : string, telefono : int)
  + modificarCliente(nombre : string, apellido : string, email : string, telefono : int)
  + eliminarCliente(email : string)
  
  + validarDatos()
  + consultarServicio()
  + consultarTurno()
  + consultarCliente()
  + verificarAnticipacion()
  + generarCodigo()
  + validarCodigo()
}

' ==== RELACIONES ====
Persona <|-- Cliente
Persona <|-- Profesional
Persona <|-- Administrador

Cliente "1" -- "N" Turno : solicita >
Profesional "1" -- "N" Turno : atiende >
Administrador "1" -- "N" Servicio : gestiona >
Turno "1" -- "N" Servicio : incluye >

Administrador "1" --> "1" Controlador : utiliza >

' relaciones con catalogos
Controlador --> CatalogoClientes
Controlador --> CatalogoProfesionales
Controlador --> CatalogoServicios
Controlador --> CatalogoTurnos

CatalogoClientes --> Cliente
CatalogoProfesionales --> Profesional
CatalogoServicios --> Servicio
CatalogoTurnos --> Turno

@enduml
