skinparam style strictuml

' =========================
' ==== CLASES BASE ========
' =========================

class Persona {
  - nombre : String
  - apellido : String
  - email : String
  - telefono : int
}

class Cliente {
  + solicitarTurno(s : Servicio, p : Profesional, fechaHora : DateTime) : Turno
  + cancelarTurno(t : Turno) : Boolean
  + verMisTurnos() : List<Turno>
  + seleccionarTurno()

  + confirmarOperacion() : Boolean
  + cancelarOperacion() : Boolean

  + ingresarDatos() : Boolean
  + ingresarCodigo(codigo : String) : Boolean
}

class Profesional {
  - especialidad : String
  - disponibilidad : List<DateTime>


}

class Administrador {


  + confirmarOperacion()
  + cancelarOperacion()
  + ingresarDatos()
  + ingresarCodigo(codigo : int)
  + seleccionarProfesionales() 
  + actualizarDisponibilidad(fechaHora : DateTime)
}

class Servicio {
  - tipo : String
  - precio : Float
}

class Turno {
  - fechaHora : DateTime
  - estado : String

  + cambiarEstado(nuevoEstado : String)
  + verificarAnticipacion() : Boolean
}

' =========================
' ==== CATALOGOS ==========
' =========================

class CatalogoClientes {
  - clientes : List<Cliente>
  + agregarCliente(c : Cliente)
  + buscarCliente(telefono : int) : Cliente
}

class CatalogoProfesionales {
  - profesionales : List<Profesional>
  + agregarProfesional(p : Profesional)
  + eliminarProfesional(p : Profesional)
  + modificarProfesional(p : Profesional)
  + buscarProfesional(nombre : String) : Profesional
}

class CatalogoServicios {
  - servicios : List<Servicio>
  + agregarServicio(s : Servicio)
  + eliminarServicio(s : Servicio)
  + modificarServicio(s : Servicio)
  + buscarServicio(tipo : String) : Servicio
}

class CatalogoTurnos {
  - turnos : List<Turno>
  + agregarTurno(t : Turno)
  + eliminarTurno(t : Turno)
  + buscarTurno(fechaHora : DateTime) : Turno
}

' =========================
' ==== CONTROLADOR ========
' =========================

class Controlador {
  - catalogoClientes : CatalogoClientes
  - catalogoProfesionales : CatalogoProfesionales
  - catalogoServicios : CatalogoServicios
  - catalogoTurnos : CatalogoTurnos

  + agregarCliente(c : Cliente)

  + agregarProfesional(p : Profesional)
  + eliminarProfesional(p : Profesional)
  + modificarProfesional(p : Profesional)

  + agregarServicio(s : Servicio)
  + eliminarServicio(s : Servicio)
  + modificarServicio(s : Servicio)

  + crearTurno(c : Cliente, p : Profesional, s : Servicio, fechaHora : DateTime) : Turno
  + cancelarTurno(t : Turno)

  + validarDatos() : Boolean
  + enviarCodigo() : String
  + validarCodigo(codigo : String) : Boolean
  + enviarConfirmacion()
  + liberarHorario(fechaHora : DateTime)

  + buscarCliente(telefono : int) : Cliente
  + buscarTurno(fechaHora : DateTime) : Turno
  + buscarServicio(tipo : String) : Servicio
  + buscarProfesional(nombre : String) : Profesional
  
}

' =========================
' ==== RELACIONES =========
' =========================

Persona <|-- Cliente
Persona <|-- Profesional
Persona <|-- Administrador

Cliente "1" -- "0..*" Turno : solicita >
Profesional "1" -- "0..*" Turno : atiende >
Turno "1" -- "1..*" Servicio : incluye >

CatalogoClientes "1" o-- "0..*" Cliente
CatalogoProfesionales "1" o-- "0..*" Profesional
CatalogoServicios "1" o-- "0..*" Servicio
CatalogoTurnos "1" o-- "0..*" Turno

Administrador --> Controlador
Controlador --> CatalogoClientes
Controlador --> CatalogoProfesionales
Controlador --> CatalogoServicios
Controlador --> CatalogoTurnos

@enduml
