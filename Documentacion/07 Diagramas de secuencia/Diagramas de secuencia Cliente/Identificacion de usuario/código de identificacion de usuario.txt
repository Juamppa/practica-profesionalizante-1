@startuml
title Identificación de Usuario

actor Usuario
participant "controlador : Controlador" as controlador
participant "cliente : Cliente" as cliente

Usuario -> controlador : enviarDatos(nombre : string, apellido : string, tel : int, email : string)
activate controlador

controlador -> controlador : validarDatos(nombre, apellido, tel, email)
note right: verifica formato y existencia

alt Datos válidos
    controlador -> cliente : agregarCliente (nombre : string, apellido : string, tel : int, email : string)
    activate cliente
    
    cliente --> controlador : <<clienteCreado>>
    deactivate cliente
    
    controlador -> Usuario : <<accesoConcedido>>

else Datos inválidos
    controlador -> Usuario : <<error: "Mail inválido o teléfono inexistente">>
end

deactivate controlador
@enduml
