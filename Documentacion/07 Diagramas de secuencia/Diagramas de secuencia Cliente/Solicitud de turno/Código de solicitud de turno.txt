@startuml
title Solicitud de Turno

actor Usuario
participant "controlador : Controlador" as controlador
participant "turno : Turno" as turno

Usuario -> controlador : seleccionarServicio(servicio : string)
activate controlador

Usuario -> controlador : seleccionarFechaHora(fechaHora : DateTime)

Usuario -> controlador : seleccionarProfesional(profesional : string)

controlador -> turno : agregarTurno (fechaHora : DateTime, estado : string, servicio : string, profesional : string)
activate turno

turno -> turno : agregarTurno()


turno --> controlador : <<turnoCreado>>

deactivate turno

controlador -> Usuario : <<mensaje: "Turno confirmado correctamente">>

deactivate controlador
@enduml
