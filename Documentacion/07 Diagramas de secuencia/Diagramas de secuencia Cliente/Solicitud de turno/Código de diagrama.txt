@startuml
actor Cliente
participant "Interfaz" as Web
participant "Controlador" as Controlador
participant "Servicio" as Servicio
participant "Profesional" as Profesional
participant "Turno" as Turno

Cliente -> Web: Acceder a interfaz principal
activate Web

Web -> Controlador: solicitarServicios()
activate Controlador
Controlador -> Servicio: obtenerServiciosDisponibles()
activate Servicio
Servicio --> Controlador: Lista de servicios
deactivate Servicio
Controlador --> Web: Mostrar servicios
deactivate Controlador

Web -> Cliente: Mostrar lista de servicios
Cliente -> Web: Seleccionar servicio

Web -> Controlador: solicitarFechasHoras(servicio)
activate Controlador
Controlador -> Turno: obtenerFechasHorasDisponibles(servicio)
activate Turno
Turno --> Controlador: Fechas y horas disponibles
deactivate Turno
Controlador --> Web: Mostrar fechas y horas
deactivate Controlador

Web -> Cliente: Mostrar fechas y horas disponibles
Cliente -> Web: Seleccionar fecha y hora

Web -> Controlador: solicitarProfesionales(servicio)
activate Controlador
Controlador -> Profesional: obtenerProfesionalesDisponibles(servicio)
activate Profesional
Profesional --> Controlador: Lista de profesionales
deactivate Profesional
Controlador --> Web: Mostrar profesionales
deactivate Controlador

Web -> Cliente: Mostrar lista de profesionales
Cliente -> Web: Seleccionar profesional

Web -> Controlador: generarCodigoVerificacion(cliente)
activate Controlador
Controlador -> Cliente: enviarCodigoWhatsapp(codigo)
deactivate Controlador

Web -> Cliente: Notificar envío de código de verificación
deactivate Web
@enduml
