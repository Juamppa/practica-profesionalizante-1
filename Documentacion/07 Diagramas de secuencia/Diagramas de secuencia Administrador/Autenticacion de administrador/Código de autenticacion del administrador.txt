@startuml
title  Autenticación del administrador

' ==== PARTICIPANTES ====
actor Administrador
participant "Sistema" as Sistema
participant "Controlador" as Controlador
participant "Administrador" as AdmObj

' ==== SECUENCIA ====

Administrador -> Sistema : enviarTelefono(telefono : int)
activate Sistema

Sistema -> Controlador : verificarTelefono(telefono : int)
activate Controlador

Controlador -> AdmObj : getTelefonoAdministrador()
activate AdmObj
AdmObj --> Controlador : telefonoAdministrador <<retorno>>
deactivate AdmObj

Controlador --> Sistema : telefonoValido() <<retorno>>
deactivate Controlador

alt Teléfono válido
    Sistema -> Controlador : generarCodigo()
    activate Controlador
    Controlador --> Sistema : codigoGenerado(codigo : int) <<retorno>>
    deactivate Controlador

    Sistema -> Administrador : enviarCodigoWhatsapp(codigo : int)
    deactivate Sistema

    Administrador -> Sistema : ingresarCodigo(codigoIngresado : int)
    activate Sistema

    Sistema -> Controlador : validarCodigo(codigoIngresado : int)
    activate Controlador
    Controlador --> Sistema : codigoCorrecto() <<retorno>>
    deactivate Controlador

    Sistema -> Administrador : accesoConcedido()
    deactivate Sistema
else Teléfono inválido
    Sistema -> Administrador : error("El número ingresado no es correcto") <<retorno>>
    deactivate Sistema
end

@enduml
