@startuml
title Modificar Profesional

actor Administrador
participant "controlador : Controlador" as controlador
participant "profesional : Profesional" as profesional

' 1. El administrador ingresa los datos modificados
Administrador -> controlador : ingresarDatosEditados(nombre : string, apellido : string, especialidad : string)
activate controlador

' 2. Búsqueda del profesional
controlador -> profesional : buscarProfesional()
activate profesional
profesional --> controlador : <<profesionalEncontrado>>
deactivate profesional

' 3. Modificación del profesional
controlador -> profesional : modificarProfesional(nombre : string, apellido : string, especialidad : string)
activate profesional

profesional -> profesional : actualizarDatos()
note right: actualiza los atributos\ncon los nuevos valores

profesional --> controlador : <<profesionalModificado>>

deactivate profesional

' 4. Mensaje de éxito al administrador
controlador -> Administrador : <<mensaje: "Profesional modificado correctamente">>

deactivate controlador
@enduml
