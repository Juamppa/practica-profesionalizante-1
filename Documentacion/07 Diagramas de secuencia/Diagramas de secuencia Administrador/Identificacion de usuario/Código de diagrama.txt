@startuml
actor Administrador
participant "Interfaz" as Web
participant "Controlador" as Controlador
participant "Administrador" as ObjAdmin

Administrador -> Web: Ingresar a la página
activate Web

Web -> Administrador: Mostrar opciones (Cliente / Administrador)
Administrador -> Web: Seleccionar opción Administrador

Web -> Administrador: Solicitar número de teléfono
Administrador -> Web: Ingresar número de teléfono

Web -> Controlador: validarTelefono(numeroTelefono)
activate Controlador
Controlador -> ObjAdmin: << create >> Administrador(telefonoAdministrador)
activate ObjAdmin
ObjAdmin --> Controlador: Teléfono cargado
deactivate ObjAdmin

Controlador -> Controlador: verificarTelefonoRegistrado()
Controlador --> Web: Resultado validación
deactivate Controlador

alt Teléfono correcto
    Web -> Controlador: generarCodigoVerificacion(Administrador)
    activate Controlador
    Controlador -> Administrador: enviarCodigoWhatsapp(codigo)
    deactivate Controlador
    
    Web -> Administrador: Solicitar ingreso del código recibido
    Administrador -> Web: Ingresar código
    
    Web -> Controlador: validarCodigo(codigo)
    activate Controlador
    Controlador -> Controlador: verificarCodigo()
    Controlador --> Web: Resultado validación
    deactivate Controlador

    alt Código correcto
        Web -> Administrador: Accede a interfaz principal del administrador
    else Código incorrecto
        Web -> Administrador: Mostrar mensaje “El código ingresado no es correcto”
        Web -> Administrador: Volver al paso de ingreso de teléfono
    end

else Teléfono incorrecto
    Web -> Administrador: Mostrar mensaje “El número de teléfono ingresado no es correcto”
    Web -> Administrador: Volver al paso de ingreso de teléfono
end

deactivate Web
@enduml
