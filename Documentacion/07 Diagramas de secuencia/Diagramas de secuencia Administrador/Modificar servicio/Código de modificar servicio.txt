@startuml
title Modificar Servicio

actor Administrador
participant "controlador : Controlador" as controlador
participant "servicio : Servicio" as servicio

' 1. El administrador ingresa los datos modificados
Administrador -> controlador : ingresarDatosEditados(tipo : string, precio : float)
activate controlador

' 2. Búsqueda del servicio
controlador -> servicio : consultarServicio()
activate servicio
servicio --> controlador : <<servicioEncontrado>>
deactivate servicio

' 3. Modificación del servicio
controlador -> servicio : modificarServicio(tipo : string, precio : float)
activate servicio

servicio -> servicio : actualizarDatos()


servicio --> controlador : <<servicioModificado>>

deactivate servicio

' 4. Mensaje de éxito al administrador
controlador -> Administrador : <<mensaje: "Servicio modificado correctamente">>

deactivate controlador
@enduml
