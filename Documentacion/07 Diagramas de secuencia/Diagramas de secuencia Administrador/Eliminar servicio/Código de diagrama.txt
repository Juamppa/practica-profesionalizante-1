@startuml
actor Administrador
participant "Interfaz" as Web
participant "Controlador" as Controlador
participant "Servicio" as Servicio


Administrador -> Web: Ingresar a la página
activate Web

Web -> Administrador: Mostrar opciones (Cliente / Administrador)
Administrador -> Web: Seleccionar opción Administrador

Web -> Administrador: Mostrar opciones ("Agregar", "Eliminar", "Modificar")
Administrador -> Web: Seleccionar opción "Eliminar"

Web -> Administrador: Mostrar opciones ("Eliminar servicio", "Eliminar profesional")
Administrador -> Web: Seleccionar "Eliminar servicio"

Web -> Controlador: obtenerServicios()
activate Controlador
Controlador -> Servicio: listarServicios()
activate Servicio
Servicio --> Controlador: Lista de servicios cargados
deactivate Servicio
Controlador --> Web: Enviar lista de servicios
deactivate Controlador

Web -> Administrador: Mostrar lista de servicios disponibles
Administrador -> Web: Seleccionar servicio(s) y presionar "Eliminar"

alt Servicio seleccionado
    Web -> Controlador: eliminarServicio(servicioSeleccionado)
    activate Controlador
    Controlador -> Servicio: eliminar()
    activate Servicio
    Servicio --> Controlador: Eliminación confirmada
    deactivate Servicio
    Controlador --> Web: Confirmación de eliminación
    deactivate Controlador
    Web -> Administrador: Mostrar mensaje “Servicio eliminado con éxito”

else Ningún servicio seleccionado
    Web -> Administrador: Mostrar mensaje “Debe seleccionar al menos un servicio para eliminar.”
end

deactivate Web
@enduml
