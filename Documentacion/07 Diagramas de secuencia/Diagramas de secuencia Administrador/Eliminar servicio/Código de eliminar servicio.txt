@startuml
title Eliminar Servicio

actor Administrador
participant "controlador : Controlador" as controlador
participant "servicio : Servicio" as servicio

Administrador -> controlador : solicitarEliminacionServicio(tipo : string, precio : float)
activate controlador

controlador -> servicio : eliminarServicio()
activate servicio

servicio --> controlador : <<servicioEliminado>>

deactivate servicio

controlador -> Administrador : <<mensaje: "Servicio eliminado correctamente">>

deactivate controlador
@enduml
