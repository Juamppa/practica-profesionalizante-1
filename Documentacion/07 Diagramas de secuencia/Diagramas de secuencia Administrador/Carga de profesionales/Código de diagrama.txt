@startuml
actor Administrador
participant "Interfaz" as Web
participant "Controlador" as Controlador
participant "Profesional" as Profesional


Administrador -> Web: Ingresar a la página
activate Web

Web -> Administrador: Mostrar opciones (Cliente / Administrador)
Administrador -> Web: Seleccionar opción Administrador

Web -> Administrador: Mostrar opciones ("Agregar", "Eliminar", "Modificar")
Administrador -> Web: Seleccionar opción "Agregar"

Web -> Administrador: Mostrar opciones ("Agregar servicio", "Agregar profesional")
Administrador -> Web: Seleccionar "Agregar profesional"

Web -> Administrador: Solicitar datos del profesional (Nombre, Apellido, Disponibilidad)
Administrador -> Web: Ingresar datos y presionar "Aceptar"

Web -> Controlador: crearProfesional(nombre, apellido, disponibilidad)
activate Controlador
Controlador -> Profesional: << create >> Profesional(nombre, apellido, disponibilidad)
activate Profesional
Profesional --> Controlador: Profesional creado
deactivate Profesional
Controlador --> Web: Confirmación de carga
deactivate Controlador

alt Campos completos
    Web -> Administrador: Mostrar mensaje “Profesional agregado con éxito”
else Campo vacío
    Web -> Administrador: Mostrar mensaje “Hay un campo vacío, por favor rellénelo para continuar”
end

deactivate Web
@enduml
