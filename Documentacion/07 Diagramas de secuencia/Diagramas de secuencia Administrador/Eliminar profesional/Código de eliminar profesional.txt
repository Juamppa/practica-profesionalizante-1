@startuml
title Eliminar Profesional

actor Administrador
participant "controlador : Controlador" as controlador
participant "profesional : Profesional" as profesional

Administrador -> controlador : solicitarEliminacionProfesional(idProfesional : int)
activate controlador

controlador -> profesional : eliminarProfesional()
activate profesional

profesional --> controlador : <<profesionalEliminado>>

deactivate profesional

controlador -> Administrador : <<mensaje: "Profesional eliminado correctamente">>

deactivate controlador
@enduml
