participant "ctrl:Controlador" as Controlador
participant "cp:CatalogoProfesionales" as Catalogo

    Controlador -> Catalogo : buscarProfesionales()
    activate Catalogo
    Catalogo --> Controlador : <<List<Profesional>>>

Admin -> Controlador : seleccionarProfesionales()


    alt ningún profesional seleccionado

    else profesionales seleccionados
        
        Admin <<-- Controlador : <<"¿Está seguro de eliminar los profesionales seleccionados?":string>>
        
        alt Administrador cancela
            
            Admin -> Controlador : cancelarOperacion()
            Admin <<-- Controlador : <<"operacion cancelada":string>>
        
        else confirma eliminación
            
            Admin -> Controlador : confirmarOperacion()
            
            
                loop para cada profSel en profsSeleccionados
                    Controlador -> Catalogo : eliminarProfesional(profSel)
                    activate Catalogo
                    Catalogo --> Controlador
                    deactivate Catalogo
                end
                
            Admin <<-- Controlador : <<"Profesional(es) eliminado(s) correctamente":string>>
            deactivate Controlador
        
        end
        
    end

deactivate Controlador
